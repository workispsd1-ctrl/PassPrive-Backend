CREATE TABLE public.restaurants (
  id uuid NOT NULL DEFAULT gen_random_uuid (),
  name text NOT NULL,
  phone text NULL,
  area text NULL,
  city text NULL,
  full_address text NULL,
  slug text NULL,
  cover_image text NULL,
  latitude numeric(9, 6) NULL,
  longitude numeric(9, 6) NULL,
  description text NULL,
  cost_for_two integer NULL,
  is_active boolean NOT NULL DEFAULT true,
  owner_user_id uuid NULL,
  is_pure_veg boolean NOT NULL DEFAULT false,
  booking_enabled boolean NOT NULL DEFAULT true,
  avg_duration_minutes integer NOT NULL DEFAULT 90,
  max_bookings_per_slot integer NULL,
  advance_booking_days integer NOT NULL DEFAULT 30,
  modification_available boolean NOT NULL DEFAULT false,
  modification_cutoff_minutes integer NULL,
  cancellation_available boolean NOT NULL DEFAULT false,
  cancellation_cutoff_minutes integer NULL,
  cover_charge_enabled boolean NOT NULL DEFAULT false,
  cover_charge_amount numeric(10, 2) NULL,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  updated_at timestamp with time zone NOT NULL DEFAULT now(),
  is_advertised boolean NOT NULL DEFAULT false,
  ad_priority integer NULL,
  ad_starts_at timestamp with time zone NULL,
  ad_ends_at timestamp with time zone NULL,
  ad_badge_text text NULL,
  booking_terms text[] NOT NULL DEFAULT '{}'::text[],
  menu_json jsonb NULL,
  google_place_id text NULL,
  source text NULL DEFAULT 'google'::text,
  source_payload jsonb NULL,
  last_synced_at timestamp without time zone NULL,
  rating numeric(2, 1) NULL,
  user_ratings_total integer NULL,
  place_types text[] NULL,
  country text NULL DEFAULT 'Mauritius'::text,
  
  CONSTRAINT restaurants_pkey PRIMARY KEY (id),
  CONSTRAINT restaurants_slug_key UNIQUE (slug),
  CONSTRAINT restaurants_google_place_id_key UNIQUE (google_place_id),
  CONSTRAINT restaurants_owner_user_id_fkey FOREIGN KEY (owner_user_id) REFERENCES auth.users (id) ON DELETE SET NULL,
  CONSTRAINT restaurants_avg_duration_minutes_chk CHECK ((avg_duration_minutes > 0)),
  CONSTRAINT restaurants_cancellation_cutoff_chk CHECK (
    ((cancellation_available = true) OR (cancellation_cutoff_minutes IS NULL))
  ),
  CONSTRAINT restaurants_cost_for_two_chk CHECK (
    ((cost_for_two IS NULL) OR (cost_for_two >= 0))
  ),
  CONSTRAINT restaurants_cover_charge_chk CHECK (
    (
      ((cover_charge_enabled = false) AND ((cover_charge_amount IS NULL) OR (cover_charge_amount = (0)::numeric)))
      OR
      ((cover_charge_enabled = true) AND (cover_charge_amount IS NOT NULL) AND (cover_charge_amount >= (0)::numeric))
    )
  ),
  CONSTRAINT restaurants_ad_consistency_chk CHECK (
    (((is_advertised = false) OR ((is_advertised = true) AND (ad_priority IS NOT NULL))))
  ),
  CONSTRAINT restaurants_latitude_chk CHECK (
    (((latitude IS NULL) OR ((latitude >= ('-90'::integer)::numeric) AND (latitude <= 90))))
  ),
  CONSTRAINT restaurants_longitude_chk CHECK (
    (((longitude IS NULL) OR ((longitude >= ('-180'::integer)::numeric) AND (longitude <= 180))))
  ),
  CONSTRAINT restaurants_max_bookings_per_slot_chk CHECK (
    (((max_bookings_per_slot IS NULL) OR (max_bookings_per_slot > 0)))
  ),
  CONSTRAINT restaurants_modification_cutoff_chk CHECK (
    (((modification_available = true) OR (modification_cutoff_minutes IS NULL)))
  ),
  CONSTRAINT restaurants_ad_priority_chk CHECK (
    (((ad_priority IS NULL) OR (ad_priority >= 0)))
  ),
  CONSTRAINT restaurants_advance_booking_days_chk CHECK ((advance_booking_days >= 0)),
  CONSTRAINT restaurants_ad_time_range_chk CHECK (
    (((ad_starts_at IS NULL) OR (ad_ends_at IS NULL) OR (ad_starts_at <= ad_ends_at)))
  )
) TABLESPACE pg_default;

-- Indices:
CREATE INDEX IF NOT EXISTS restaurants_active_city_area_idx ON public.restaurants USING btree (is_active, city, area) TABLESPACE pg_default;
CREATE INDEX IF NOT EXISTS restaurants_owner_user_id_idx ON public.restaurants USING btree (owner_user_id) TABLESPACE pg_default;
CREATE INDEX IF NOT EXISTS restaurants_location_idx ON public.restaurants USING btree (latitude, longitude) TABLESPACE pg_default;
CREATE INDEX IF NOT EXISTS restaurants_advertised_idx ON public.restaurants USING btree (is_advertised, ad_priority, ad_starts_at, ad_ends_at) TABLESPACE pg_default;

-- Triggers:
CREATE TRIGGER trg_restaurants_set_updated_at BEFORE UPDATE ON restaurants FOR EACH ROW EXECUTE FUNCTION set_updated_at ();



CREATE TABLE public.restaurant_reviews (
  id uuid NOT NULL DEFAULT gen_random_uuid (),
  restaurant_id uuid NOT NULL,
  user_id uuid NULL, -- NULL for external reviews
  rating numeric(2, 1) NOT NULL,
  review_text text NULL,
  liked_tags text[] NOT NULL DEFAULT '{}'::text[],
  photo_urls text[] NOT NULL DEFAULT '{}'::text[],
  username_snapshot text NULL,
  avatar_snapshot text NULL,
  is_approved boolean NOT NULL DEFAULT true,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  updated_at timestamp with time zone NOT NULL DEFAULT now(),
  food_rating numeric(2, 1) NULL,
  service_rating numeric(2, 1) NULL,
  ambience_rating numeric(2, 1) NULL,
  drinks_rating numeric(2, 1) NULL,
  crowd_rating numeric(2, 1) NULL,
  owner_reply_text text NULL,
  owner_reply_by uuid NULL,
  owner_reply_at timestamp with time zone NULL,
  owner_reply_updated_at timestamp with time zone NULL,
  source text NULL DEFAULT 'internal'::text,
  external_review_id text NULL,
  
  CONSTRAINT restaurant_reviews_pkey PRIMARY KEY (id),
  CONSTRAINT restaurant_reviews_user_id_fkey FOREIGN KEY (user_id) REFERENCES auth.users (id) ON DELETE CASCADE,
  CONSTRAINT restaurant_reviews_owner_reply_by_fkey FOREIGN KEY (owner_reply_by) REFERENCES auth.users (id) ON DELETE SET NULL,
  CONSTRAINT restaurant_reviews_restaurant_id_fkey FOREIGN KEY (restaurant_id) REFERENCES restaurants (id) ON DELETE CASCADE,
  CONSTRAINT restaurant_reviews_service_rating_chk CHECK (
    ((service_rating IS NULL) OR ((service_rating >= (0)::numeric) AND (service_rating <= (5)::numeric)))
  ),
  CONSTRAINT restaurant_reviews_crowd_rating_chk CHECK (
    ((crowd_rating IS NULL) OR ((crowd_rating >= (0)::numeric) AND (crowd_rating <= (5)::numeric)))
  ),
  CONSTRAINT restaurant_reviews_rating_chk CHECK (
    (((rating >= (0)::numeric) AND (rating <= (5)::numeric)))
  ),
  CONSTRAINT restaurant_reviews_ambience_rating_chk CHECK (
    ((ambience_rating IS NULL) OR ((ambience_rating >= (0)::numeric) AND (ambience_rating <= (5)::numeric)))
  ),
  CONSTRAINT restaurant_reviews_drinks_rating_chk CHECK (
    ((drinks_rating IS NULL) OR ((drinks_rating >= (0)::numeric) AND (drinks_rating <= (5)::numeric)))
  ),
  CONSTRAINT restaurant_reviews_food_rating_chk CHECK (
    ((food_rating IS NULL) OR ((food_rating >= (0)::numeric) AND (food_rating <= (5)::numeric)))
  )
) TABLESPACE pg_default;

-- Indices:
CREATE INDEX IF NOT EXISTS restaurant_reviews_restaurant_id_idx ON public.restaurant_reviews USING btree (restaurant_id) TABLESPACE pg_default;
CREATE INDEX IF NOT EXISTS restaurant_reviews_user_id_idx ON public.restaurant_reviews USING btree (user_id) TABLESPACE pg_default;
CREATE INDEX IF NOT EXISTS restaurant_reviews_created_at_idx ON public.restaurant_reviews USING btree (created_at DESC) TABLESPACE pg_default;
CREATE INDEX IF NOT EXISTS restaurant_reviews_liked_tags_gin_idx ON public.restaurant_reviews USING gin (liked_tags) TABLESPACE pg_default;
CREATE INDEX IF NOT EXISTS restaurant_reviews_owner_reply_at_idx ON public.restaurant_reviews USING btree (owner_reply_at DESC) TABLESPACE pg_default;

-- Triggers:
CREATE TRIGGER set_restaurant_reviews_updated_at BEFORE UPDATE ON restaurant_reviews FOR EACH ROW EXECUTE FUNCTION set_updated_at ();
CREATE TRIGGER trg_set_review_reply_timestamps BEFORE UPDATE ON restaurant_reviews FOR EACH ROW EXECUTE FUNCTION set_review_reply_timestamps ();

CREATE TABLE public.restaurant_opening_hours (
  id uuid NOT NULL DEFAULT gen_random_uuid (),
  restaurant_id uuid NOT NULL,
  day_of_week integer NOT NULL,
  open_time time without time zone NULL,
  close_time time without time zone NULL,
  is_closed boolean NOT NULL DEFAULT false,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  updated_at timestamp with time zone NOT NULL DEFAULT now(),
  
  CONSTRAINT restaurant_opening_hours_pkey PRIMARY KEY (id),
  CONSTRAINT restaurant_opening_hours_restaurant_id_fkey FOREIGN KEY (restaurant_id) REFERENCES restaurants (id) ON DELETE CASCADE,
  CONSTRAINT restaurant_opening_hours_day_of_week_chk CHECK (
    ((day_of_week >= 0) AND (day_of_week <= 6))
  ),
  CONSTRAINT restaurant_opening_hours_time_order_chk CHECK (
    ((is_closed = true) OR (open_time <> close_time))
  ),
  CONSTRAINT restaurant_opening_hours_time_required_chk CHECK (
    (
      ((is_closed = true) AND (open_time IS NULL) AND (close_time IS NULL))
      OR
      ((is_closed = false) AND (open_time IS NOT NULL) AND (close_time IS NOT NULL))
    )
  )
) TABLESPACE pg_default;

-- Indices:
CREATE INDEX IF NOT EXISTS restaurant_opening_hours_restaurant_id_idx ON public.restaurant_opening_hours USING btree (restaurant_id) TABLESPACE pg_default;
CREATE INDEX IF NOT EXISTS restaurant_opening_hours_restaurant_day_idx ON public.restaurant_opening_hours USING btree (restaurant_id, day_of_week) TABLESPACE pg_default;
CREATE UNIQUE INDEX IF NOT EXISTS restaurant_opening_hours_restaurant_day_unique_idx ON public.restaurant_opening_hours USING btree (restaurant_id, day_of_week) TABLESPACE pg_default;

-- Triggers:
CREATE TRIGGER trg_restaurant_opening_hours_set_updated_at BEFORE UPDATE ON restaurant_opening_hours FOR EACH ROW EXECUTE FUNCTION set_updated_at ();


CREATE TABLE public.restaurant_media_assets (
  id uuid NOT NULL DEFAULT gen_random_uuid (),
  restaurant_id uuid NOT NULL,
  asset_type text NOT NULL,
  file_url text NOT NULL,
  file_path text NULL,
  sort_order integer NOT NULL DEFAULT 100,
  is_active boolean NOT NULL DEFAULT true,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  updated_at timestamp with time zone NOT NULL DEFAULT now(),
  google_photo_reference text NULL,
  local_file_path text NULL,
  mime_type text NULL,
  storage_bucket text NULL,
  storage_path text NULL,
  storage_public_url text NULL,
  enhanced_file_url text NULL,
  enhanced_storage_path text NULL,
  enhanced_storage_public_url text NULL,
  
  CONSTRAINT restaurant_media_assets_pkey PRIMARY KEY (id),
  CONSTRAINT restaurant_media_assets_restaurant_id_fkey FOREIGN KEY (restaurant_id) REFERENCES restaurants (id) ON DELETE CASCADE,
  CONSTRAINT restaurant_media_assets_asset_type_chk CHECK (
    ((asset_type = ANY (ARRAY['food'::text, 'ambience'::text, 'menu'::text, 'cover'::text])))
  ),
  CONSTRAINT restaurant_media_assets_file_url_chk CHECK (
    ((length(TRIM(BOTH FROM file_url)) > 0))
  ),
  CONSTRAINT restaurant_media_assets_sort_order_chk CHECK ((sort_order >= 0))
) TABLESPACE pg_default;

-- Indices:
CREATE INDEX IF NOT EXISTS restaurant_media_assets_restaurant_id_idx ON public.restaurant_media_assets USING btree (restaurant_id) TABLESPACE pg_default;
CREATE INDEX IF NOT EXISTS restaurant_media_assets_asset_type_idx ON public.restaurant_media_assets USING btree (asset_type) TABLESPACE pg_default;
CREATE INDEX IF NOT EXISTS restaurant_media_assets_restaurant_type_idx ON public.restaurant_media_assets USING btree (restaurant_id, asset_type) TABLESPACE pg_default;
CREATE INDEX IF NOT EXISTS restaurant_media_assets_restaurant_active_sort_idx ON public.restaurant_media_assets USING btree (restaurant_id, is_active, sort_order, created_at) TABLESPACE pg_default;
CREATE UNIQUE INDEX IF NOT EXISTS restaurant_media_assets_one_cover_per_restaurant_idx ON public.restaurant_media_assets USING btree (restaurant_id) TABLESPACE pg_default WHERE ((asset_type = 'cover'::text) AND (is_active = true));
CREATE UNIQUE INDEX IF NOT EXISTS restaurant_media_assets_rest_photo_uq ON public.restaurant_media_assets USING btree (restaurant_id, google_photo_reference) TABLESPACE pg_default WHERE (google_photo_reference IS NOT NULL);

-- Triggers:
CREATE TRIGGER trg_restaurant_media_assets_set_updated_at BEFORE UPDATE ON restaurant_media_assets FOR EACH ROW EXECUTE FUNCTION set_updated_at ();
